// When Appointed Status is updated to Completed, then billing will automatically 
created with Name, Fees and Payment date. 

trigger AppointmentCompletedBillingCreated on Appointment__c (after update) {
    
    List<Billing__c> billst = new List<Billing__c>();
    
    for(Appointment__c appt : Trigger.new){
        
        Appointment__c appold = Trigger.oldMap.get(appt.Id);
        
        if(appt.Status__c == 'Completed' && appold.Status__c != 'Completed'){
            Billing__c bill = new Billing__c();
            bill.Name = appt.Name + ' Billing';
            bill.Appointments__c = appt.Id;
            bill.Consultation_Fee__c = 500;
            bill.Payment_Date__c = Date.today();
            bill.Status__c = 'Pending';
            billst.add(bill);
            
        }
 
    }
    if(!billst.isEmpty()){
        insert billst;
    }
    
}
